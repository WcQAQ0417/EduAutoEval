# 教务自动填表 - 方案一：浏览器控制台脚本

## 方案概述
这是最简单的方案，无需安装任何软件或扩展。只需要在浏览器中打开教务系统，按F12打开开发者工具，粘贴脚本即可运行。

## 优点
- ✅ 零安装，无需任何额外软件
- ✅ 使用简单，复制粘贴即可
- ✅ 直接在当前浏览器会话中运行，复用已登录状态
- ✅ 代码透明可见，安全可靠

## 缺点
- ❌ 每次使用都需要手动打开控制台粘贴脚本
- ❌ 页面刷新后脚本需要重新运行
- ❌ 功能相对简单

## 使用步骤

### 第一步：登录教务系统
1. 打开浏览器，访问 https://jws.qgxy.cn
2. 使用你的账号密码登录系统
3. 导航到教学评估列表页面：https://jws.qgxy.cn/student/teachingEvaluation/evaluation/index

### 第二步：打开浏览器开发者工具
- Windows/Linux: 按 `F12` 或 `Ctrl + Shift + I`
- Mac: 按 `Cmd + Option + I`
- 或者右键点击页面，选择"检查"或"Inspect"

### 第三步：切换到Console（控制台）标签
点击开发者工具中的"Console"标签页

### 第四步：粘贴并运行脚本
将以下脚本复制粘贴到控制台中，按回车运行

---

## 自动化脚本代码

```javascript
// ==========================================
// 教务系统自动评教脚本 - 控制台版
// 使用方法：在评估列表页按F12，粘贴到Console运行
// ==========================================

(function() {
    'use strict';
    
    // 配置项
    const CONFIG = {
        // 评分策略：'good'=全部选好/较好, 'random'=随机评分, 'mixed'=混合评分
        scoreStrategy: 'good',
        // 提交前等待时间（毫秒），会自动跳过2分钟倒计时
        submitDelay: 3000,
        // 主观评价内容（会随机选择一条）
        subjectiveComments: [
            '老师教学认真负责，讲解清晰易懂，课堂氛围很好，收获很大',
            '老师备课充分，重点突出，能够激发学生学习兴趣',
            '教学方法得当，注重互动，能够很好地引导学生思考',
            '老师授课生动有趣，理论联系实际，学到了很多知识',
            '老师责任心强，答疑耐心，课程设计合理，非常满意'
        ],
        // 是否自动跳过已完成的评估
        skipCompleted: true,
        // 每完成一个评估后等待时间（毫秒）
        nextEvaluationDelay: 2000
    };
    
    let isRunning = true;
    let completedCount = 0;
    
    // 在页面上显示状态面板
    function createStatusPanel() {
        const panel = document.createElement('div');
        panel.id = 'auto-eval-panel';
        panel.style.cssText = `
            position: fixed;
            top: 20px;
            right: 20px;
            width: 300px;
            background: white;
            border: 2px solid #438EB9;
            border-radius: 8px;
            padding: 15px;
            z-index: 999999;
            box-shadow: 0 4px 20px rgba(0,0,0,0.2);
            font-family: -apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif;
            font-size: 14px;
        `;
        panel.innerHTML = `
            <div style="font-weight: bold; font-size: 16px; color: #438EB9; margin-bottom: 10px; border-bottom: 1px solid #eee; padding-bottom: 8px;">
                🤖 自动评教助手
            </div>
            <div id="eval-status" style="margin-bottom: 8px;">状态：准备中...</div>
            <div id="eval-progress" style="margin-bottom: 8px;">进度：0 / 0</div>
            <div id="eval-current" style="margin-bottom: 10px; color: #666; font-size: 12px;">当前：-</div>
            <button id="eval-stop" style="width: 100%; padding: 8px; background: #d15b47; color: white; border: none; border-radius: 4px; cursor: pointer;">
                停止自动评教
            </button>
        `;
        document.body.appendChild(panel);
        
        document.getElementById('eval-stop').addEventListener('click', () => {
            isRunning = false;
            updateStatus('已停止');
        });
    }
    
    function updateStatus(text) {
        const el = document.getElementById('eval-status');
        if (el) el.textContent = '状态：' + text;
    }
    
    function updateProgress(current, total) {
        const el = document.getElementById('eval-progress');
        if (el) el.textContent = `进度：${current} / ${total}`;
    }
    
    function updateCurrent(text) {
        const el = document.getElementById('eval-current');
        if (el) el.textContent = '当前：' + text;
    }
    
    // 获取未评估的列表项
    function getUnevaluatedItems() {
        // 查找所有"评估"按钮（紫色按钮）
        const buttons = Array.from(document.querySelectorAll('button.btn-purple, button.btn-success')).filter(btn => {
            const text = btn.textContent.trim();
            return text.includes('评估') && !text.includes('查看') && !text.includes('修改');
        });
        return buttons;
    }
    
    // 自动填写评估页面
    function fillEvaluationForm() {
        updateStatus('正在填写问卷...');
        
        // 1. 填写所有单选题
        const radioGroups = {};
        
        // 收集所有单选题组
        document.querySelectorAll('input[type="radio"]').forEach(radio => {
            const name = radio.name;
            if (!radioGroups[name]) {
                radioGroups[name] = [];
            }
            radioGroups[name].push(radio);
        });
        
        // 为每组选择答案
        Object.keys(radioGroups).forEach((name, index) => {
            const options = radioGroups[name];
            if (options.length === 0) return;
            
            let selectedIndex;
            
            switch (CONFIG.scoreStrategy) {
                case 'good':
                    // 选第一个或第二个选项（好/较好，完全符合/符合）
                    selectedIndex = Math.random() > 0.3 ? 0 : 1;
                    break;
                case 'random':
                    // 完全随机
                    selectedIndex = Math.floor(Math.random() * Math.min(options.length, 3));
                    break;
                case 'mixed':
                default:
                    // 混合策略：大部分选好的，偶尔选较好或一般
                    const rand = Math.random();
                    if (rand > 0.8) {
                        selectedIndex = 2; // 一般
                    } else if (rand > 0.4) {
                        selectedIndex = 1; // 较好/符合
                    } else {
                        selectedIndex = 0; // 好/完全符合
                    }
            }
            
            // 确保索引有效
            selectedIndex = Math.min(selectedIndex, options.length - 1);
            
            // 点击选中
            setTimeout(() => {
                options[selectedIndex].click();
                options[selectedIndex].checked = true;
                // 触发change事件
                options[selectedIndex].dispatchEvent(new Event('change', { bubbles: true }));
            }, index * 50);
        });
        
        // 2. 填写主观评价
        setTimeout(() => {
            const textarea = document.querySelector('textarea[name="zgpj"]');
            if (textarea) {
                const comment = CONFIG.subjectiveComments[Math.floor(Math.random() * CONFIG.subjectiveComments.length)];
                textarea.value = comment;
                textarea.dispatchEvent(new Event('input', { bubbles: true }));
                textarea.dispatchEvent(new Event('change', { bubbles: true }));
            }
            
            updateStatus('填写完成，准备提交...');
            
            // 3. 绕过2分钟倒计时，直接提交
            setTimeout(() => {
                submitForm();
            }, CONFIG.submitDelay);
            
        }, Object.keys(radioGroups).length * 50 + 500);
    }
    
    // 提交表单
    function submitForm() {
        updateStatus('正在提交...');
        
        // 绕过时间限制
        if (typeof flag !== 'undefined') {
            flag = true;
        }
        
        // 尝试直接调用提交函数
        if (typeof toEvaluation === 'function') {
            // 先确认弹窗（如果有的话）
            const originalConfirm = window.confirm;
            const originalLayerConfirm = layer.confirm;
            
            // 覆盖layer.confirm自动确认
            layer.confirm = function(msg, options, callback) {
                if (typeof callback === 'function') {
                    callback(true);
                }
                return 0;
            };
            
            // 覆盖urp.alert避免弹窗阻塞
            if (typeof urp !== 'undefined' && urp.alert) {
                const originalAlert = urp.alert;
                urp.alert = function(msg) {
                    console.log('系统提示:', msg);
                    // 如果是成功消息，跳回列表页
                    if (msg.includes('成功')) {
                        setTimeout(() => {
                            window.location.href = '/student/teachingEvaluation/evaluation/index';
                        }, 500);
                    }
                };
            }
            
            try {
                toEvaluation('submit');
            } catch (e) {
                console.error('提交出错，尝试直接提交表单:', e);
                document.StDaForm.submit();
            }
            
            // 恢复原始函数
            setTimeout(() => {
                layer.confirm = originalLayerConfirm;
                if (urp && originalAlert) urp.alert = originalAlert;
            }, 1000);
            
        } else {
            // 直接提交表单
            document.StDaForm.submit();
        }
    }
    
    // 检查当前页面类型并执行相应操作
    function checkPageAndRun() {
        const currentUrl = window.location.href;
        
        if (currentUrl.includes('evaluation/index') || currentUrl.includes('evaluationPage') === false) {
            // 在列表页
            runOnListPage();
        } else if (currentUrl.includes('evaluationPage') || document.querySelector('form[name="StDaForm"]')) {
            // 在评估页
            runOnEvaluationPage();
        } else {
            updateStatus('未知页面，正在跳转到列表页...');
            setTimeout(() => {
                window.location.href = '/student/teachingEvaluation/evaluation/index';
            }, 1000);
        }
    }
    
    // 在列表页运行
    function runOnListPage() {
        // 等待列表加载完成
        const checkList = setInterval(() => {
            const evalButtons = getUnevaluatedItems();
            
            if (evalButtons.length > 0) {
                clearInterval(checkList);
                
                updateProgress(completedCount, completedCount + evalButtons.length);
                updateStatus(`找到 ${evalButtons.length} 个未评估项目`);
                
                // 点击第一个评估按钮
                setTimeout(() => {
                    const btn = evalButtons[0];
                    const row = btn.closest('tr');
                    const teacherName = row ? row.querySelectorAll('td')[2]?.textContent?.trim() : '未知';
                    const courseName = row ? row.querySelectorAll('td')[3]?.textContent?.trim() : '未知';
                    
                    updateCurrent(`${teacherName} - ${courseName}`);
                    updateStatus('正在进入评估页面...');
                    
                    btn.click();
                }, 1000);
            } else {
                // 检查是否有"查看"按钮（已完成）
                const viewButtons = document.querySelectorAll('button.btn-info');
                if (viewButtons.length > 0 && completedCount > 0) {
                    clearInterval(checkList);
                    updateStatus('🎉 所有评估已完成！');
                    alert(`自动评教完成！共完成 ${completedCount} 个评估。`);
                } else {
                    updateStatus('等待列表加载...');
                }
            }
        }, 500);
    }
    
    // 在评估页运行
    function runOnEvaluationPage() {
        updateStatus('已进入评估页面，等待页面加载...');
        
        // 等待表单加载完成
        const checkForm = setInterval(() => {
            const form = document.querySelector('form[name="StDaForm"]');
            if (form) {
                clearInterval(checkForm);
                fillEvaluationForm();
            }
        }, 300);
    }
    
    // 监听页面跳转完成
    function watchPageChanges() {
        let lastUrl = location.href;
        new MutationObserver(() => {
            const url = location.href;
            if (url !== lastUrl) {
                lastUrl = url;
                if (isRunning) {
                    setTimeout(checkPageAndRun, 1000);
                }
            }
        }).observe(document, { subtree: true, childList: true });
    }
    
    // 主函数
    function main() {
        console.log('%c🤖 自动评教脚本已启动', 'color: #438EB9; font-size: 16px; font-weight: bold;');
        console.log('%c配置信息:', 'color: #666;', CONFIG);
        
        createStatusPanel();
        updateStatus('脚本已启动');
        
        // 等待页面完全加载
        setTimeout(() => {
            checkPageAndRun();
            watchPageChanges();
        }, 500);
    }
    
    // 启动
    if (document.readyState === 'complete' || document.readyState === 'interactive') {
        main();
    } else {
        window.addEventListener('DOMContentLoaded', main);
    }
    
})();
```

## 使用注意事项

1. **评分策略说明**
   - `good`：所有题目选"好"或"较好"（推荐，安全）
   - `random`：随机选择前3个选项
   - `mixed`：大部分选好的，偶尔选较好或一般（更真实）

2. **主观评价**
   - 脚本内置了5条好评模板
   - 你可以自行修改 `subjectiveComments` 数组添加更多评价
   - 注意：评价不能有空格，不能有连续3个相同字符，至少4个字

3. **绕过2分钟限制**
   - 脚本会自动将 `flag` 变量设为 `true`，绕过倒计时
   - 实际等待3秒后就会提交（可通过 `submitDelay` 配置）

4. **如果脚本不工作**
   - 确保你在正确的页面（评估列表页或评估页）
   - 刷新页面后需要重新运行脚本
   - 检查控制台是否有错误信息

## 停止脚本
- 点击右上角状态面板的"停止自动评教"按钮
- 或者刷新页面即可停止
